void main()
{
	char str1[100];
	char str2[11];
	gets(str1);
	for(int i = 0; str1[i] != '\0'; i++)
	{
		if(str1[i] == 'E' || str1[i] == 'e')
		{
			strncpy(str2, &str1[i], 10);
			str2[10] = '\0';
			break;
		}
	}
	puts(str2);
}
